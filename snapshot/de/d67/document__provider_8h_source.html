40 // signed-in user. In practice this is a second request in parallel with that

59  // Returns a set of classifications that highlight all the occurrences of
60  // |input_text| at word breaks in |text|. E.g., given |input_text|
61  // "rain if you dare" and |text| "how to tell if your kitten is a rainbow",
62  // will return the classifications:
63  // __ ___ ____
64  // how to tell if your kitten is a rainbow

139  // Appends |matches_cache_| to |matches_|. Updates their classifications
140  // according to |input_.text()| and sets their relevance to 0.
141  // |skip_n_most_recent_matches| indicates the number of cached matches already
142  // in |matches_|. E.g. if the drive server responded with 3 docs, these 3
143  // docs are added both to |matches_| and |matches_cache| prior to invoking
144  // |CopyCachedMatchesToMatches()| in order to avoid duplicate matches.
145  void CopyCachedMatchesToMatches(size_t skip_n_most_recent_matches = 0);
146 
147  // Sets the scores of all cached matches to 0. This is invoked before pushing
148  // the latest async response returns so that the scores aren't preserved for
149  // further inputs. E.g., the input 'london' shouldn't display cached docs from
150  // a previous input 'paris'. This can't be done by automatically (i.e. set
151  // scores to 0 before pushing to the cache), as the scores are needed for the
152  // async pass if the user continued their input.
153  void SetCachedMatchesScoresTo0();

159  void DemoteMatchesBeyondMax();
