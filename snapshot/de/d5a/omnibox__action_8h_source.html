26 // Omnibox Actions are additional actions associated with matches. They appear
27 // in the suggestion button row and are not matches themselves.
28 //
29 // Omnibox Pedals are a type of Omnibox Action.
30 //
31 // Actions are ref-counted to support a flexible ownership model.
32 // - Some actions are ephemeral and specific to the match, and should be
33 // destroyed when the match is destroyed, so matches have the only reference.
34 // - Some actions (like Pedals) are fixed and expensive to copy, so matches
35 // should merely hold one of the references to the action.
36 class OmniboxAction : public base::RefCounted<OmniboxAction> {
37  public:

39  LabelStrings(int id_hint,
40  int id_suggestion_contents,
41  int id_accessibility_suffix,
42  int id_accessibility_hint);

44  LabelStrings(const LabelStrings&);
45  ~LabelStrings();
46  std::u16string hint;
47  std::u16string suggestion_contents;
48  std::u16string accessibility_suffix;
49  std::u16string accessibility_hint;
50  };

68  // ExecutionContext provides the necessary structure for Action
69  // execution implementations that potentially vary widely, and
70  // references are preferred over pointers for members that are
71  // not nullable. If there's ever a good case for changing to
72  // nullable pointers, this can change but for now presence is
73  // guaranteed so Actions can use them without needing to check.
74  // The other reason to use a context is that it's easier to
75  // maintain with lots of Action implementations because the amount
76  // of boilerplate required is greatly reduced.
77  class ExecutionContext {
78  public:
79  ExecutionContext(Client& client,
80  OmniboxEditController& controller,
81  base::TimeTicks match_selection_timestamp)
82  : client_(client),
83  controller_(controller),
84  match_selection_timestamp_(match_selection_timestamp) {}
85  Client& client_;
86  OmniboxEditController& controller_;
87  base::TimeTicks match_selection_timestamp_;
88  };
