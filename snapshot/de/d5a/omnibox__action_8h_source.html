31 // Omnibox Actions are additional actions associated with matches. They appear
32 // in the suggestion button row and are not matches themselves.
33 //
34 // Omnibox Pedals are a type of Omnibox Action.
35 //
36 // Actions are ref-counted to support a flexible ownership model.
37 // - Some actions are ephemeral and specific to the match, and should be
38 // destroyed when the match is destroyed, so matches have the only reference.
39 // - Some actions (like Pedals) are fixed and expensive to copy, so matches
40 // should merely hold one of the references to the action.
41 class OmniboxAction : public base::RefCounted<OmniboxAction> {
42  public:

46  int id_accessibility_suffix,

50  ~LabelStrings();
51  std::u16string hint;
52  std::u16string suggestion_contents;
53  std::u16string accessibility_suffix;
54  std::u16string accessibility_hint;
55  };

77  // ExecutionContext provides the necessary structure for Action
78  // execution implementations that potentially vary widely, and
79  // references are preferred over pointers for members that are
80  // not nullable. If there's ever a good case for changing to
81  // nullable pointers, this can change but for now presence is
82  // guaranteed so Actions can use them without needing to check.
83  // The other reason to use a context is that it's easier to
84  // maintain with lots of Action implementations because the amount
85  // of boilerplate required is greatly reduced.
86  class ExecutionContext {
87  public:
88  // Set `match_type` as if the user just typed url verbatim.
89  // `destination_url_entered_without_scheme` is used to determine whether
90  // navigations typed without a scheme and upgraded to HTTPS should fall back
91  // to HTTP. The URL might have been entered without a scheme, but Action
92  // destination URLs don't need a fallback so it's fine to pass false here.
93  using OpenUrlCallback =
94  base::OnceCallback<void(const GURL& destination_url,
95  TemplateURLRef::PostContent* post_content,
96  WindowOpenDisposition disposition,
97  ui::PageTransition transition,
98  AutocompleteMatchType::Type match_type,
99  base::TimeTicks match_selection_timestamp,
100  bool destination_url_entered_without_scheme,
101  const std::u16string&,
102  const AutocompleteMatch&,
103  const AutocompleteMatch&)>;
104 
105  ExecutionContext(Client& client,
106  OpenUrlCallback callback,
107  base::TimeTicks match_selection_timestamp,
108  WindowOpenDisposition disposition);
109  ~ExecutionContext();
110  Client& client_;
111  OpenUrlCallback open_url_callback_;
112  base::TimeTicks match_selection_timestamp_;
113  WindowOpenDisposition disposition_;
114  };
