83 // A single result line with classified spans. The autocomplete popup displays
84 // the 'contents' and the 'description' (the description is optional) in the
85 // autocomplete dropdown, and fills in 'fill_into_edit' into the textbox when
86 // that line is selected. fill_into_edit may be the same as 'description' for
87 // things like URLs, but may be different for searches or other providers. For
88 // example, a search result may say "Search for asdf" as the description, but
89 // "asdf" should appear in the box.
90 struct AutocompleteMatch {
91  // Autocomplete matches contain strings that are classified according to a
92  // separate vector of styles. This vector associates flags with particular
93  // string segments, and must be in sorted order. All text must be associated
94  // with some kind of classification. Even if a match has no distinct
95  // segments, its vector should contain an entry at offset 0 with no flags.
96  //
97  // Example: The user typed "goog"
98  // http://www.google.com/ Google
99  // ^ ^ ^ ^ ^
100  // 0, | 15, | 4,
101  // 11,match 0,match
102  //
103  // This structure holds the classification information for each span.
104  struct ACMatchClassification {
105  // The values in here are not mutually exclusive -- use them like a
106  // bit field. This also means we use "int" instead of this enum type when
107  // passing the values around, so the compiler doesn't complain.
108  //
109  // A Java counterpart will be generated for this enum.
110  // GENERATED_JAVA_ENUM_PACKAGE: org.chromium.chrome.browser.omnibox
111  // GENERATED_JAVA_CLASS_NAME_OVERRIDE: MatchClassificationStyle
112  // clang-format off
113  enum Style {
114  NONE = 0,
115  URL = 1 << 0, // A URL
116  MATCH = 1 << 1, // A match for the user's search term
117  DIM = 1 << 2, // "Helper text"
118  };
119  // clang-format on
120 
121  ACMatchClassification(size_t offset, int style)
122  : offset(offset),
123  style(style) {
124  }
125 
126  bool operator==(const ACMatchClassification& other) const {
127  return offset == other.offset && style == other.style;
128  }
129 
130  bool operator!=(const ACMatchClassification& other) const {
131  return offset != other.offset || style != other.style;
132  }
133 
134  // Offset within the string that this classification starts
135  size_t offset;
136 
137  // Contains a bitmask of flags defined in enum Style.
138  int style;
139  };

348  // Returns |url| altered by stripping off "www.", converting https protocol
349  // to http, and stripping excess query parameters. These conversions are
350  // merely to allow comparisons to remove likely duplicates; these URLs are
351  // not used as actual destination URLs.
352  // - |input| is used to decide if the scheme is allowed to be altered during
353  // stripping. If this URL, minus the scheme and separator, starts with any
354  // the terms in input.terms_prefixed_by_http_or_https(), we avoid converting
355  // an HTTPS scheme to HTTP. This means URLs that differ only by these
356  // schemes won't be marked as dupes, since the distinction seems to matter
357  // to the user.
358  // - If |template_url_service| is not NULL, it is used to get a template URL
359  // corresponding to this match, which is used to strip off query args other
360  // than the search terms themselves that would otherwise prevent doing
361  // proper deduping.
362  // - If the match's keyword is known, it can be provided in |keyword|.
363  // Otherwise, it can be left empty and the template URL (if any) is
364  // determined from the destination's hostname.
365  static GURL GURLToStrippedGURL(const GURL& url,
