24 // An HistoryMatch that has a score as well as metrics defining where in the
25 // history item's URL and/or page title matches have occurred.
26 struct ScoredHistoryMatch : public history::HistoryMatch {
27  // ScoreMaxRelevance maps from an intermediate-score to the maximum
28  // final-relevance score given to a URL for this intermediate score.
29  // This is used to store the score ranges of relevance buckets.
30  // Please see GetFinalRelevancyScore() for details.
31  using ScoreMaxRelevance = std::pair<double, int>;
32 
33  // A sorted vector of ScoreMaxRelevance entries, used by taking a score and
34  // interpolating between consecutive buckets. See GetFinalRelevancyScore()
35  // for details.
36  using ScoreMaxRelevances = std::vector<ScoreMaxRelevance>;

39  ScoredHistoryMatch();

42  ScoredHistoryMatch& operator=(const ScoredHistoryMatch& other);
43  ScoredHistoryMatch& operator=(ScoredHistoryMatch&& other);

45  // Initializes the ScoredHistoryMatch with a raw score calculated for the
46  // history item given in |row| with recent visits as indicated in |visits|. It
47  // first determines if the row qualifies by seeing if all of the terms in
48  // |terms_vector| occur in |row|. If so, calculates a raw score. This raw
49  // score is in part determined by whether the matches occur at word
50  // boundaries, the locations of which are stored in |word_starts|. For some
51  // terms, it's appropriate to look for the word boundary within the term. For
52  // instance, the term ".net" should look for a word boundary at the "n".
53  // These offsets (".net" should have an offset of 1) come from
54  // |terms_to_word_starts_offsets|. |is_url_bookmarked| indicates whether the
55  // match's URL is referenced by any bookmarks, which can also affect the raw
56  // score. |num_matching_pages| indicates how many URLs in the eligible URL
57  // database match the user's input; it can also affect the raw score. The raw
58  // score allows the matches to be ordered and can be used to influence the
59  // final score calculated by the client of this index. If the row does not
60  // qualify the raw score will be 0.
61  ScoredHistoryMatch(const history::URLRow& row,
62  const VisitInfoVector& visits,
63  const std::u16string& lower_string,
64  const String16Vector& terms_vector,
65  const WordStarts& terms_to_word_starts_offsets,
66  const RowWordStarts& word_starts,
67  bool is_url_bookmarked,
68  size_t num_matching_pages,
69  base::Time now);

71  ~ScoredHistoryMatch();

73  // Compares two matches by score. Functor supporting URLIndexPrivateData's
74  // HistoryItemsForTerms function. Looks at particular fields within
75  // with url_info to make tie-breaking a bit smarter.
76  static bool MatchScoreGreater(const ScoredHistoryMatch& m1,
77  const ScoredHistoryMatch& m2);
78 
79  // Returns |term_matches| after removing all matches that are not at a
80  // word break that are in the range [|start_pos|, |end_pos|).
81  // start_pos == string::npos is treated as start_pos = length of string.
82  // (In other words, no matches will be filtered.)
83  // end_pos == string::npos is treated as end_pos = length of string. If
84  // |allow_midword_continuations| is true, matches not at a word break are not
85  // filtered if they continue where the previous match ended.
86  static TermMatches FilterTermMatchesByWordStarts(
87  const TermMatches& term_matches,
88  const WordStarts& terms_to_word_starts_offsets,
89  const WordStarts& word_starts,
90  size_t start_pos,
91  size_t end_pos,
92  bool allow_midword_continuations = false);
93 
94  // An interim score taking into consideration location and completeness
95  // of the match.
96  int raw_score;
97 
98  // Both these TermMatches contain the set of matches that are considered
99  // important. At this time, that means they exclude mid-word matches
100  // except in the hostname of the URL. (Technically, during early
101  // construction of ScoredHistoryMatch, they may contain all matches, but
102  // unimportant matches are eliminated by GetTopicalityScore(), called
103  // during construction.)
104 
105  // Term matches within the URL.
106  TermMatches url_matches;
107  // Term matches within the page title.
108  TermMatches title_matches;

124  // Return a topicality score based on how many matches appear in the url and
125  // the page's title and where they are (e.g., at word boundaries). Revises
126  // url_matches and title_matches in the process so they only reflect matches
127  // used for scoring. (For instance, some mid-word matches are not given
128  // credit in scoring.) Requires that |url_matches| and |title_matches| are
129  // sorted. |adjustments| must contain any adjustments used to format |url|.
130  float GetTopicalityScore(const int num_terms,
131  const GURL& url,
132  const base::OffsetAdjuster::Adjustments& adjustments,
133  const WordStarts& terms_to_word_starts_offsets,
134  const RowWordStarts& word_starts);

136  // Returns a recency score based on |last_visit_days_ago|, which is
137  // how many days ago the page was last visited.
138  float GetRecencyScore(int last_visit_days_ago) const;
139 
140  // Examines the first |max_visits_to_score_| and returns a score (higher is
141  // better) based the rate of visits, whether the page is bookmarked, and
142  // how often those visits are typed navigations (i.e., explicitly
143  // invoked by the user). |now| is passed in to avoid unnecessarily
144  // recomputing it frequently.
145  float GetFrequency(const base::Time& now,
146  const bool bookmarked,
147  const VisitInfoVector& visits) const;
148 
149  // Returns a document specificity score based on how many pages matched the
150  // user's input.
151  float GetDocumentSpecificityScore(size_t num_matching_pages) const;
152 
153  // Combines the three component scores into a final score that's
154  // an appropriate value to use as a relevancy score.
155  static float GetFinalRelevancyScore(float topicality_score,
156  float frequency_score,
157  float specificity_score);

161  static ScoreMaxRelevances GetHQPBuckets();
162 
163  // Helper function to parse the string containing the scoring buckets and
164  // return the results. For example, with |buckets_str| as
165  // "0.0:400,1.5:600,12.0:1300,20.0:1399", it returns [(0.0, 400), (1.5, 600),
166  // (12.0, 1300), (20.0, 1399)]. It returns an empty vector in the case of a
167  // malformed |buckets_str|.
168  static ScoreMaxRelevances GetHQPBucketsFromString(
169  const std::string& buckets_str);

171  // If true, assign raw scores to be max(whatever it normally would be, a

173  static bool also_do_hup_like_scoring_;
174

178 
179  // Typed visits to page score this, compared to 1 for untyped visits.

181 
182  // The maximum number of recent visits to examine in GetFrequency().
183  static size_t max_visits_to_score_;
184 
185  // If true, we allow input terms to match in the TLD (e.g., ".com").
186  static bool allow_tld_matches_;
187 
188  // If true, we allow input terms to match in the scheme (e.g., "http://").
189  static bool allow_scheme_matches_;
190 
191  // The number of title words examined when computing topicality scores.
192  // Words beyond this number are ignored.
193  static size_t num_title_words_to_allow_;
194 
195  // |topicality_threshold_| is used to control the topicality scoring.
196  // If |topicality_threshold_| > 0, then URLs with topicality-score less than

198  static float topicality_threshold_;

208  static OmniboxFieldTrial::NumMatchesScores* matches_to_specificity_override_;
