42  // The Result classes are intermediate representations of AutocompleteMatches,
43  // simply containing relevance-ranked search and navigation suggestions.

45  // new suggestions from updated input are in flight.

48  // highly fragmented SearchProvider logic for each Result type.

286  // A simple structure bundling most of the information (including
287  // both SuggestResults and NavigationResults) returned by a call to
288  // the suggest server.
289  //
290  // This has to be declared after the typedefs since it relies on some of them.
291  struct Results {
292  Results();
293  ~Results();
294  Results(const Results&) = delete;
295  Results& operator=(const Results&) = delete;
296 
297  // Clears |suggest_results| and |navigation_results| and resets
298  // |verbatim_relevance| to -1 (implies unset).
299  void Clear();
300 
301  // Returns whether any of the results (including verbatim) have
302  // server-provided scores.
303  bool HasServerProvidedScores() const;
304 
305  // Query suggestions sorted by relevance score, descending. This order is
306  // normally provided by server and is guaranteed after search provider
307  // calls SortResults, so order always holds except possibly while parsing.
308  SuggestResults suggest_results;
309 
310  // Navigational suggestions sorted by relevance score, descending. This
311  // order is normally provided by server and is guaranteed after search
312  // provider calls SortResults, so order always holds except possibly while
313  // parsing.
314  NavigationResults navigation_results;
315 
316  // The server supplied verbatim relevance scores. Negative values
317  // indicate that there is no suggested score; a value of 0
318  // suppresses the verbatim result.
319  int verbatim_relevance;
320 
321  // The JSON metadata associated with this server response.
322  std::string metadata;
323 
324  // If the active suggest field trial (if any) has triggered.
325  bool field_trial_triggered;
326 
327  // The list of experiment stats which needs to be logged to SearchboxStats
328  // as part of a GWS experiment, if any.
329  ExperimentStats experiment_stats;
330 
331  // If the relevance values of the results are from the server.
332  bool relevances_from_server;
333 
334  // The server supplied map of suggestion group IDs to header labels.
335  HeadersMap headers_map;
336 
337  // The server supplied list of group IDs that should be hidden-by-default.
338  std::vector<int> hidden_group_ids;
339  };
