43 // Autocomplete provider for searches and suggestions from a search engine.
44 //
45 // After construction, the autocomplete controller repeatedly calls Start()
46 // with some user input, each time expecting to receive a small set of the best
47 // matches (either synchronously or asynchronously).
48 //
49 // Initially the provider creates a match that searches for the current input
50 // text. It also starts a task to query the Suggest servers. When that data
51 // comes back, the provider creates and returns matches for the best
52 // suggestions.

113  // Manages the providers (TemplateURLs) used by SearchProvider. Two providers
114  // may be used:
115  // . The default provider. This corresponds to the user's default search
116  // engine. This is always used, except for the rare case of no default
117  // engine.
118  // . The keyword provider. This is used if the user has typed in a keyword.
119  class Providers {
120  public:
121  explicit Providers(TemplateURLService* template_url_service);
122  Providers(const Providers&) = delete;
123  Providers& operator=(const Providers&) = delete;
124 
125  // Returns true if the specified providers match the two providers cached
126  // by this class.
127  bool equal(const std::u16string& default_provider,
128  const std::u16string& keyword_provider) const {
129  return (default_provider == default_provider_) &&
130  (keyword_provider == keyword_provider_);
131  }
132 
133  // Resets the cached providers.
134  void set(const std::u16string& default_provider,
135  const std::u16string& keyword_provider) {
136  default_provider_ = default_provider;
137  keyword_provider_ = keyword_provider;
138  }
139 
140  const std::u16string& default_provider() const { return default_provider_; }
141  const std::u16string& keyword_provider() const { return keyword_provider_; }
142 
143  // NOTE: These may return NULL even if the provider members are nonempty!
144  const TemplateURL* GetDefaultProviderURL() const;
145  const TemplateURL* GetKeywordProviderURL() const;
146 
147  // Returns true if there is a valid keyword provider.
148  bool has_keyword_provider() const { return !keyword_provider_.empty(); }
149 
150  private:
151  TemplateURLService* template_url_service_;
152 
153  // Cached across the life of a query so we behave consistently even if the
154  // user changes their default while the query is running.
155  std::u16string default_provider_;
156  std::u16string keyword_provider_;
157  };
