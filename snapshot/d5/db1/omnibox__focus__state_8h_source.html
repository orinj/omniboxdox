26 // Reasons why the Omnibox focus state could change.
27 enum OmniboxFocusChangeReason {
28  // Includes any explicit changes to focus. (e.g. user clicking to change
29  // focus, user tabbing to change focus, any explicit calls to SetFocus,
30  // etc.)
31  OMNIBOX_FOCUS_CHANGE_EXPLICIT,

33  // Focus changed to restore state from a tab the user switched to.
