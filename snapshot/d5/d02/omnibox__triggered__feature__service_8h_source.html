10 // Tracks the features that trigger during an omnibox session and records them
11 // to the logs. This is used for counterfactual slicing metrics by feature.
12 class OmniboxTriggeredFeatureService {
13  public:
14  // The list of features used for counterfactual slicing.
15  // These values are persisted to logs. Entries should not be renumbered and
16  // numeric values should never be reused. When adding an entry here, a
17  // corresponding entry should be added in the UMA histograms.
18  enum class Feature {
19  kRichAutocompletion = 0,
20  kBookmarkPaths = 1,
21  kShortBookmarkSuggestionsByTotalInputLength = 2,
22  kMaxValue = kShortBookmarkSuggestionsByTotalInputLength,
23  };
24  using Features = std::set<Feature>;
25 
26  OmniboxTriggeredFeatureService();
27  ~OmniboxTriggeredFeatureService();
28 
29  // Records |features_| for this session to |feature_triggered_in_session|.
30  void RecordToLogs(Features* feature_triggered_in_session) const;
31 
32  // Invoked to indicate |feature| was triggered.
33  void TriggerFeature(Feature feature);
34 
35  // Invoked when a new omnibox session starts. Clears |features_|.
36  void ResetSession();
37 
38  private:
39  // The set of features triggered in the current omnibox session via
40  // |TriggerFeature()|.
41  std::set<Feature> features_;
42 };
