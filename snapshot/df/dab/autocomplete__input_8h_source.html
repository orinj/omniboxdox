22 // The user input for an autocomplete query. Allows copying.
23 class AutocompleteInput {
24  public:
25  AutocompleteInput();
26  // |text| represents the input query.
27  //
28  // |current_page_classification| represents the type of page the user is
29  // viewing and manner in which the user is accessing the omnibox; it's
30  // more than simply the URL. It includes, for example, whether the page
31  // is a search result page doing search term replacement or not.

33  // |scheme_classifier| is passed to Parse() to help determine the type of
34  // input this is; see comments there.

43  // query |text|. It may be set to std::u16string::npos if the input
44  // doesn't come directly from the user's typing.

73  // Parses |text| (including an optional |desired_tld|) and returns the type of
74  // input this will be interpreted as. |scheme_classifier| is used to check
75  // the scheme in |text| is known and registered in the current environment.
76  // The components of the input are stored in the output parameter |parts|, if
77  // it is non-NULL. The scheme is stored in |scheme| if it is non-NULL. The
78  // canonicalized URL is stored in |canonicalized_url|; however, this URL is
79  // not guaranteed to be valid, especially if the parsed type is, e.g., QUERY.
80  static metrics::OmniboxInputType Parse(
81  const std::u16string& text,
82  const std::string& desired_tld,
83  const AutocompleteSchemeClassifier& scheme_classifier,
84  url::Parsed* parts,
85  std::u16string* scheme,
86  GURL* canonicalized_url);

88  // Parses |text| and fill |scheme| and |host| by the positions of them.
89  // The results are almost as same as the result of Parse(), but if the scheme
90  // is view-source, this function returns the positions of scheme and host
91  // in the URL qualified by "view-source:" prefix.
92  static void ParseForEmphasizeComponents(
93  const std::u16string& text,
94  const AutocompleteSchemeClassifier& scheme_classifier,
95  url::Component* scheme,
96  url::Component* host);

98  // Returns true if the given text and url combination should be upgraded to
99  // use https:// as the default scheme. If so, fills |upgraded_url| with the
100  // upgraded https:// URL. |https_port_for_testing| can be set to a non-zero
101  // value in tests to load test cases over net::EmbeddedTestServer.
102  static bool ShouldUpgradeToHttps(const std::u16string& text,
103  const GURL& url,
104  int https_port_for_testing,
105  GURL* upgraded_url);

107  // Code that wants to format URLs with a format flag including
108  // net::kFormatUrlOmitTrailingSlashOnBareHostname risk changing the meaning if
109  // the result is then parsed as AutocompleteInput. Such code can call this
110  // function with the URL and its formatted string, and it will return a
111  // formatted string with the same meaning as the original URL (i.e. it will
112  // re-append a slash if necessary). Because this uses Parse() under the hood
113  // to determine the meaning of the different strings, callers need to supply a
114  // |scheme_classifier| to pass to Parse(). If |offset| is non-null, it will
115  // be updated with any changes that shift it.
116  static std::u16string FormattedStringWithEquivalentMeaning(
117  const GURL& url,

120  size_t* offset);

199  // |prefer_keyword| should be true when the keyword UI is onscreen; this

204  }

206  // Returns whether this input is allowed to be treated as an exact
207  // keyword match. If not, the default result is guaranteed not to be a

212  // highlighting text in a page and telling the browser to search for it or
213  // navigate to it. This member only applies to substituting keywords.
