27 // Used by ZeroSuggestProvider only. This does duplicate some logic found in
28 // SearchProvider already. As-you-type suggest does NOT use this.
29 //
30 // TODO(tommycli): This class used to be much larger. At this point, it may make
31 // sense to to just fold it into ZeroSuggestProvider - or rename this class to
32 // include ZeroSuggest in the name.
33 //
34 // A service to fetch suggestions from the default search provider's suggest
35 // service.
36 //
37 // This service is always sent the user's authentication state, so the
38 // suggestions always can be personalized. This service is also sometimes sent
39 // the user's current URL, so the suggestions are sometimes also contextual.
40 class RemoteSuggestionsService : public KeyedService {
41  public:
42  explicit RemoteSuggestionsService(
43  scoped_refptr<network::SharedURLLoaderFactory> url_loader_factory);
44 
45  ~RemoteSuggestionsService() override;
46  RemoteSuggestionsService(const RemoteSuggestionsService&) = delete;

49  using StartCallback = base::OnceCallback<void(
50  std::unique_ptr<network::SimpleURLLoader> loader)>;
51 
52  using CompletionCallback =
53  base::OnceCallback<void(const network::SimpleURLLoader* source,
54  std::unique_ptr<std::string> response_body)>;
55 
56  // Returns a URL representing the address of the server where the zero suggest
57  // request is being sent. Does not take into account whether sending this
58  // request is prohibited (e.g. in an incognito window).
59  // Returns an invalid URL (i.e.: GURL::is_valid() == false) in case of an
60  // error.
61  //
62  // |search_terms_args| encapsulates the arguments sent to the suggest service.
63  // Various parts of it (including the current page URL and classification) are
64  // used to build the final endpoint URL. Note that the current page URL can
65  // be empty.
66  //
67  // Note that this method is public and is also used by ZeroSuggestProvider for
68  // suggestions that do not take the current page URL into consideration.
69  static GURL EndpointUrl(TemplateURLRef::SearchTermsArgs search_terms_args,
70  const TemplateURLService* template_url_service);
71 
72  // Creates a loader for remote suggestions for |search_terms_args| and passes
73  // the loader to |start_callback|. It uses a number of signals to create the
74  // loader, including field trial / experimental parameters, and it may
75  // pass a nullptr to |start_callback| (see below for details).
76  //
77  // |search_terms_args| encapsulates the arguments sent to the remote service.
78  // If |search_terms_args.current_page_url| is empty, the system will never use
79  // the experimental suggestions service. It's possible the non-experimental
80  // service may decide to offer general-purpose suggestions.
81  //
82  // |template_url_service| may be null, but some services may be disabled.
83  //
84  // |start_callback| is called to transfer ownership of the created loader to
85  // whatever function/class receives the callback.
86  //
87  // |completion_callback| will be invoked when the transfer is done.
88  void CreateSuggestionsRequest(
89  const TemplateURLRef::SearchTermsArgs& search_terms_args,
90  const TemplateURLService* template_url_service,
91  StartCallback start_callback,
92  CompletionCallback completion_callback);
93 
94  private:
95  // Activates a loader for |request|, wiring it up to |completion_callback|,
96  // and calls |start_callback|. If |request_body| isn't empty, it will be
97  // attached as upload bytes.
98  void StartDownloadAndTransferLoader(
99  std::unique_ptr<network::ResourceRequest> request,
100  std::string request_body,
101  net::NetworkTrafficAnnotationTag traffic_annotation,
102  StartCallback start_callback,
103  CompletionCallback completion_callback);
104 
105  scoped_refptr<network::SharedURLLoaderFactory> url_loader_factory_;
106 };
