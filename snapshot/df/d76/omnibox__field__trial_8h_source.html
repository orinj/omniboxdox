29 // The set of parameters customizing the HUP scoring.
30 struct HUPScoringParams {
31  // A set of parameters describing how to cap a given count score. First,
32  // we apply a half-life based decay of the given count and then find the
33  // maximum relevance score based on the decay factor or counts specified
34  // in the corresponding bucket list. See comment on |buckets_| for details.
35  class ScoreBuckets {
36  public:
37  // Stores the max relevance at each count/decay factor threshold.
38  typedef std::pair<double, int> CountMaxRelevance;
39 
40  ScoreBuckets();
41  ScoreBuckets(const ScoreBuckets& other);
42  ~ScoreBuckets();
43 
44  // Computes a half-life time decay given the |elapsed_time|.
45  double HalfLifeTimeDecay(const base::TimeDelta& elapsed_time) const;
46 
47  int relevance_cap() const { return relevance_cap_; }
48  void set_relevance_cap(int relevance_cap) {
49  relevance_cap_ = relevance_cap;
50  }
51 
52  int half_life_days() const { return half_life_days_; }
53  void set_half_life_days(int half_life_days) {
54  half_life_days_ = half_life_days;
55  }
56 
57  bool use_decay_factor() const { return use_decay_factor_; }
58  void set_use_decay_factor(bool use_decay_factor) {
59  use_decay_factor_ = use_decay_factor;
60  }
61 
62  std::vector<CountMaxRelevance>& buckets() { return buckets_; }
63  const std::vector<CountMaxRelevance>& buckets() const { return buckets_; }
64 
65  // Estimates dynamic memory usage.
66  // See base/trace_event/memory_usage_estimator.h for more info.
67  size_t EstimateMemoryUsage() const;
68 
69  private:
70  // History matches with relevance score greater or equal to |relevance_cap_|
71  // are not affected by this experiment.
72  // Set to -1, if there is no relevance cap in place and all matches are
73  // subject to demotion.
74  int relevance_cap_;
75 
76  // Half life time for a decayed count as measured since the last visit.
77  // Set to -1 if not used.
78  int half_life_days_;
79 
80  // The relevance score caps at successively decreasing threshold values.
81  // The thresholds are either decayed counts or decay factors, depending on
82  // the value of |use_decay_factor_|.
83  //
84  // Consider this example specifying the decayed counts:
85  // [(1, 1000), (0.5, 500), (0, 100)]
86  // If decayed count is 2 (which is >= 1), the corresponding match's maximum
87  // relevance will be capped at 1000. In case of 0.5, the score is capped
88  // at 500. Anything below 0.5 is capped at 100.
89  //
90  // This list is sorted by the pair's first element in descending order.
91  std::vector<CountMaxRelevance> buckets_;
92 
93  // True when the bucket thresholds are decay factors rather than counts.
94  bool use_decay_factor_;
95  };
96 
97  HUPScoringParams() {}
98 
99  // Estimates dynamic memory usage.
100  // See base/trace_event/memory_usage_estimator.h for more info.
101  size_t EstimateMemoryUsage() const;
102 
103  ScoreBuckets typed_count_buckets;
104 
105  // Used only when the typed count is 0.
106  ScoreBuckets visited_count_buckets;
107 };

117 // specificity score used in HistoryQuick provider / ScoredHistoryMatch
118 // scoring. The vector is sorted by the size_t (the number of matching pages).
119 // If an entry is omitted, the appropriate value is assumed to be the one in
120 // the later bucket. For example, with a vector containing {{1, 2.0},
121 // {3, 1.5}}, the score for 2 is inferred to be 1.5. Values beyond the
122 // end of the vector are assumed to have scores of 1.0.

577 namespace internal {
578 // The bundled omnibox experiment comes with a set of parameters
579 // (key-value pairs). Each key indicates a certain rule that applies in
580 // a certain context. The value indicates what the consequences of
581 // applying the rule are. For example, the value of a SearchHistory rule
582 // in the context of a search results page might indicate that we should
