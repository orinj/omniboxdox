21 // Note: This is not an autocomplete provider; it doesn't produce suggestions
22 // but rather "annotates" suggestions by attaching pedals to matches from other
23 // providers (search in particular).

97  // Contains mapping from well-known identifier to Pedal implementation.
98  // Note: since the set is small, we use one map here for simplicity; but if
99  // someday there are lots of Pedals, it may make sense to switch this to a
100  // vector and index by id separately. The lookup is needed rarely but
101  // iterating over the whole collection happens very frequently, so we should
102  // really optimize for iteration (vector), not lookup (map).
103  std::unordered_map<OmniboxPedalId, scoped_refptr<OmniboxPedal>> pedals_;
104 
105  // Common words that may be used when typing to trigger Pedals. All instances
106  // of these words are removed from match text when looking for triggers.
107  // Therefore no Pedal should have a trigger or synonym group that includes
108  // any of these words (as a whole word; substrings are fine).
109  OmniboxPedal::SynonymGroup ignore_group_;
110 
111  // Map from string token to unique int token identifier.
112  std::unordered_map<std::u16string, int> dictionary_;
113 
114  // This contains all token delimiter characters. It may be empty, in which

116  std::u16string tokenize_characters_;
117 
118  // This holds the tokens currently being matched against.
119  OmniboxPedal::TokenSequence match_tokens_;
120 
121  // This serves as an upper bound on the number of tokens we will accept from
122  // text before giving up and treating it as non-match for all Pedals.
123  size_t max_tokens_ = 0;
124 
125  // Whether a field trial has triggered for this query and this session
126  bool field_trial_triggered_ = false;
127  bool field_trial_triggered_in_session_ = false;
