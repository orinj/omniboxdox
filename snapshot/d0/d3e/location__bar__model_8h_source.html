21 // This class provides information about the current navigation entry.
22 // Its methods always return data related to the current page, and does not
23 // account for the state of the omnibox, which is tracked by OmniboxEditModel.
24 class LocationBarModel {
25  public:
26  virtual ~LocationBarModel() = default;
27 
28  // Returns the formatted full URL for the toolbar. The formatting includes:
29  // - Some characters may be unescaped.
30  // - The scheme and/or trailing slash may be dropped.
31  // This method specifically keeps the URL suitable for editing by not
32  // applying any elisions that change the meaning of the URL.
33  virtual std::u16string GetFormattedFullURL() const = 0;
34 
35  // Returns a simplified URL for display (but not editing) on the toolbar.
36  // This formatting is generally a superset of GetFormattedFullURL, and may
37  // include some destructive elisions that change the meaning of the URL.
38  // The returned string is not suitable for editing, and is for display only.
39  virtual std::u16string GetURLForDisplay() const = 0;
40 
41  // Returns the URL of the current navigation entry.
42  virtual GURL GetURL() const = 0;
43 
44  // Returns the security level that the toolbar should display.
45  virtual security_state::SecurityLevel GetSecurityLevel() const = 0;
46 
47  // Returns the cert status of the current navigation entry.
48  virtual net::CertStatus GetCertStatus() const = 0;
49 
50  // Classify the current page being viewed as, for example, the new tab
51  // page or a normal web page. Used for logging omnibox events for
52  // UMA opted-in users. Examines the user's profile to determine if the
53  // current page is the user's home page.
54  virtual metrics::OmniboxEventProto::PageClassification GetPageClassification(
55  OmniboxFocusSource focus_source) = 0;
56 
57  // Returns the id of the icon to show to the left of the address, based on the
58  // current URL. When search term replacement is active, this returns a search
59  // icon.
60  virtual const gfx::VectorIcon& GetVectorIcon() const = 0;
61 
62  // Returns text for the omnibox secure verbose chip, displayed next to the
63  // security icon on certain platforms.
64  virtual std::u16string GetSecureDisplayText() const = 0;
65 
66  // Returns text describing the security state for accessibility.
67  virtual std::u16string GetSecureAccessibilityText() const = 0;
68 
69  // Returns whether the URL for the current navigation entry should be
70  // in the location bar.
71  virtual bool ShouldDisplayURL() const = 0;
72 
73  // Returns whether the page is an offline page, sourced from a cache of
74  // previously-downloaded content.
75  virtual bool IsOfflinePage() const = 0;
76 
77  // Returns whether to prevent elision of the display URL, based on whether
78  // user has a specified extension or pref enabled. If true, the only elisions
79  // should be username/password and trailing slash on bare hostname.
80  virtual bool ShouldPreventElision() const = 0;
81 
82  protected:
83  LocationBarModel() = default;
84 
85  LocationBarModel(const LocationBarModel&) = delete;
86  LocationBarModel& operator=(const LocationBarModel&) = delete;
87 };
