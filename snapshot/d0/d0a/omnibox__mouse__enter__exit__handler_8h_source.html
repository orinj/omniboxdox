17 // Omnibox suggestion rows (and header rows) rely on mouse-enter and mouse-exit
18 // events to update their hover state (whether or not they are highlighted).
19 // But child Views sometimes interfere with receiving these events when the
20 // mouse is moved quickly, which causes rows to remain erroneously highlighted.
21 // This class solves that problem by bridging the child Views mouse event
22 // handler to a callback. This class instance must outlive all observed Views.
23 class OmniboxMouseEnterExitHandler : public ui::EventHandler {
24  public:
25  // |enter_exit_callback| is called whenever one of the observed Views get a
26  // mouse-enter or mouse-exit event.
27  explicit OmniboxMouseEnterExitHandler(
28  base::RepeatingClosure enter_exit_callback);
29  OmniboxMouseEnterExitHandler(const OmniboxMouseEnterExitHandler&) = delete;
30  OmniboxMouseEnterExitHandler& operator=(const OmniboxMouseEnterExitHandler&) =
31  delete;
32  ~OmniboxMouseEnterExitHandler() override;
33 
34  // This instance must outlive |view|.
35  void ObserveMouseEnterExitOn(views::View* view);
36 
37  private:
38  void OnMouseEvent(ui::MouseEvent* event) override;
39 
40  // This is called whenever one of the |observed_views_| has a mouse-enter or
41  // mouse-exit event.
42  const base::RepeatingClosure enter_exit_callback_;
43 
44  // These are the Views for which we are observing mouse-enter or mouse-exit
45  // events. This instance must outlive all of these Views, since these are
46  // non-owning pointers, which we use in our destructor.
47  std::vector<views::View*> observed_views_;
48 };
