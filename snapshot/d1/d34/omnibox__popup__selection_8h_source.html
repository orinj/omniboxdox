13 struct OmniboxPopupSelection {
14  // Directions for stepping through selections. These may apply for going
15  // up/down by lines or cycling left/right through states within a line.
16  enum Direction { kForward, kBackward };
17 
18  // When changing selections, these are the possible stepping behaviors.
19  enum Step {
20  // Step by an entire line regardless of line state. Usually used for the
21  // Up and Down arrow keys.
22  kWholeLine,
23 
24  // Step by a state if another one is available on the current line;
25  // otherwise step by line. Usually used for the Tab and Shift+Tab keys.
26  kStateOrLine,
27 
28  // Step across all lines to the first or last line. Usually used for the
29  // PgUp and PgDn keys.
30  kAllLines
31  };
32 
33  // See `Selection::state` below for details. The numeric values are to aid
34  // comparison only. They are not persisted anywhere and can be freely changed.
35  enum LineState {
36  // This means the Header above this row is highlighted, and the
37  // header collapse/expand button is focused.
38  FOCUSED_BUTTON_HEADER = 0,
39 
40  // NORMAL means the row is focused, and Enter key navigates to the match.
41  NORMAL = 1,
42 
43  // KEYWORD_MODE state is used when in Keyword mode. If the keyword search
44  // button is enabled, keyword mode is entered when the keyword button is
45  // focused.
46  KEYWORD_MODE = 2,
47 
48  // FOCUSED_BUTTON_TAB_SWITCH state means the Switch Tab button is focused.
49  // Pressing enter will switch to the tab match.
50  FOCUSED_BUTTON_TAB_SWITCH = 3,
51 
52  // FOCUSED_BUTTON_ACTION state means an Action button (such as a Pedal)
53  // is in focus.
54  FOCUSED_BUTTON_ACTION = 4,
55 
56  // FOCUSED_BUTTON_REMOVE_SUGGESTION state means the Remove Suggestion (X)
57  // button is focused. Pressing enter will attempt to remove this suggestion.
58  FOCUSED_BUTTON_REMOVE_SUGGESTION = 5,
59 
60  // Whenever new line state is added, accessibility label for current
61  // selection should be revisited
62  // (OmniboxEditModel::GetAccessibilityLabelForCurrentSelection).
63  LINE_STATE_MAX_VALUE
64  };
65 
66  // The sentinel value for Selection::line which means no line is selected.
67  static const size_t kNoMatch;
68 
69  // The selected line. This is kNoMatch when nothing is selected,
70  // which should only be true when the popup is closed.
71  size_t line;
72 
73  // If the selected line has both a normal match and a keyword match, this
74  // determines whether the normal match (if NORMAL) or the keyword match
75  // (if KEYWORD) is selected. Likewise, if the selected line has a normal
76  // match and a secondary button match, this determines whether the button
77  // match (if FOCUSED_BUTTON_*) is selected.
78  LineState state;
79 
80  explicit OmniboxPopupSelection(size_t line, LineState state = NORMAL)
81  : line(line), state(state) {}
82 
83  bool operator==(const OmniboxPopupSelection&) const;
84  bool operator!=(const OmniboxPopupSelection&) const;
85  bool operator<(const OmniboxPopupSelection&) const;
86 
87  // Returns true if going to this selection from given `from` selection
88  // results in activation of keyword state when it wasn't active before.
89  bool IsChangeToKeyword(OmniboxPopupSelection from) const;
90 
91  // Returns true if this selection represents a button being focused.
92  bool IsButtonFocused() const;
93 
94  // Returns true if the control represented by this selection's `state` is
95  // present on the match for `line` in given `result`.
96  bool IsControlPresentOnMatch(const AutocompleteResult& result,
97  PrefService* pref_service) const;
98 
99  // Returns the next selection after this one in given `result`.
100  OmniboxPopupSelection GetNextSelection(const AutocompleteResult& result,
101  PrefService* pref_service,
102  Direction direction,
103  Step step) const;
104 
105  private:
106  // This is a utility function to support `GetNextSelection`.
107  static std::vector<OmniboxPopupSelection> GetAllAvailableSelectionsSorted(
108  const AutocompleteResult& result,
109  PrefService* pref_service,
110  Direction direction,
111  Step step);
112 };
