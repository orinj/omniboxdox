19 // Conceptually, a Pedal is a fixed action that can be taken by the user
20 // pressing a button or taking a new dedicated suggestion when some
21 // associated intention is detected in an omnibox match suggestion.
22 // The typical action is to navigate to an internal Chrome surface
23 // like settings, but other actions like translation or launching
24 // an incognito window are possible. The intention is detected by
25 // checking trigger queries against suggested match queries.
26 class OmniboxPedal : public OmniboxAction {
27  public:
28  struct Token {
29  // Token identifier from the common token dictionary.
30  int id;
31 
32  // Index of the next unconsumed token node. Initially this is set to
33  // the token's own index, indicating this token is unconsumed. Calls
34  // to |TokenSequence::Consume| may then update it to greater values,
35  // indicating that this token is consumed.
36  size_t link;
37  };
38 
39  // This is a specialized container for the sequence matching algorithm.
40  // It is intended only for single-threaded access by OmniboxPedal classes.
41  class TokenSequence {
42  public:
43  // Construct with reserved size; used when loading real data.
44  explicit TokenSequence(size_t reserve_size);
45 
46  // Construct with given sequence of |token_ids|; used by tests.
47  explicit TokenSequence(std::vector<int> token_ids);
48 
49  // Don't use copies. They were necessary with old algorithm,
50  // but this structure is amenable to efficient resets on kept instances.
51  TokenSequence(const TokenSequence&) = delete;
52  TokenSequence& operator=(const TokenSequence&) = delete;
53  TokenSequence(TokenSequence&&);
54  TokenSequence& operator=(TokenSequence&&);
55  ~TokenSequence();
56 
57  // Returns true if all tokens are consumed (true for empty sequences).
58  bool IsFullyConsumed();
59 
60  // Returns the number of unconsumed tokens remaining. Used by tests.
61  size_t CountUnconsumed() const;
62 
63  // Add token with given |id| to sequence.
64  void Add(int id);
65 
66  // Clears all tokens from this sequence.
67  inline void Clear() { tokens_.clear(); }
68 
69  // Initializes all links in sequence to their own index, indicating
70  // unconsumed state for all. This is needed after calls to Erase.
71  void ResetLinks();
72 
73  // Removes one or more instances of |erase_sequence| from this sequence
74  // by erasing token items from the |tokens_| container.
75  // Returns true if this sequence was changed; false if no match is found.
76  bool Erase(const TokenSequence& erase_sequence, bool erase_only_once);
77 
78  // Consumes one or more instances of |consume_sequence| from this sequence
79  // by updating links on matching tokens. The container is not modified,
80  // only its contained elements may be mutated.
81  // Returns true if matches were found and consumed; false if no match.
82  bool Consume(const TokenSequence& consume_sequence, bool consume_only_once);
83 
84  // Returns the total number of tokens, regardless of consumed status.
85  inline size_t Size() const { return tokens_.size(); }
86 
87  // Returns collection memory estimate for tracing.
88  size_t EstimateMemoryUsage() const;
89 
90  private:
91  // Returns true if this sequence, starting at |index|, matches given
92  // |sequence|. The |index_mask| can be used to disregard consumed status (0)
93  // or require that all tokens must be unconsumed to match (~0).
94  bool MatchesAt(const TokenSequence& sequence,
95  size_t index,
96  size_t index_mask) const;
97 
98  // Follows links on tokens starting at |from_index| and returns the first
99  // unconsumed index. Returns |Size()| if no unconsumed token is found.
100  size_t WalkToUnconsumedIndexFrom(size_t from_index);
101 
102  // Storage for tokens.
103  std::vector<Token> tokens_;
104  };

112  class SynonymGroup {
113  public:
114  // Note: synonyms must be specified in decreasing order by length
115  // so that longest matches will be detected first. For example,
116  // "incognito window" must come before "incognito" so that the " window"
117  // part will also be covered by this group -- otherwise it would be left
118  // intact and wrongly treated as uncovered by the checking algorithm.
119  // See OmniboxPedal::IsConceptMatch for the logic that necessitates order.
120  SynonymGroup(bool required, bool match_once, size_t reserve_size);
121  SynonymGroup(SynonymGroup&&);
122  ~SynonymGroup();
123  SynonymGroup(const SynonymGroup&) = delete;
124  SynonymGroup& operator=(const SynonymGroup&) = delete;
125  SynonymGroup& operator=(SynonymGroup&&);
126 
127  // Removes one or more matching synonyms from given |remaining| sequence if
128  // any are found. Returns true if checking may continue; false if no more
129  // checking is required because what remains cannot be a concept match.
130  // Note, if |fully_erase| is true and this method returns true, the
131  // |remaining| container has changed structure so ResetLinks must be called.
132  // This method doesn't call ResetLinks in that case, for efficiency.
133  bool EraseMatchesIn(TokenSequence& remaining, bool fully_erase) const;
134 
135  // Add a synonym token sequence to this group.
136  void AddSynonym(TokenSequence synonym);
137 
138  // When runtime data was preprocessed by pedal_processor,
139  // it avoided the need to sort at runtime in Chromium, but with
140  // the TC-based l10n technique, data loading needs to be robust
141  // enough to handle various forms and orders in translation data.
142  // Hence, a call to `SortSynonyms` is required after all calls
143  // to `AddSynonym` are complete. We may eliminate this step
144  // if we implement post-processing of the .xtb translation files.
145  void SortSynonyms();
146 
147  // Estimates RAM usage in bytes for this synonym group.
148  size_t EstimateMemoryUsage() const;
149 
150  // Erases sequences in ignore group from all synonyms in this group.
151  void EraseIgnoreGroup(const SynonymGroup& ignore_group);
152 
153  // Returns true if this synonym group contains nontrivial data that can
154  // be used by the matching algorithm.
155  bool IsValid() const;
156 
157  protected:
158  // If this is true, a synonym of the group must be present for triggering.
159  // If false, then presence is simply allowed and does not inhibit triggering
160  // (any text not covered by groups would stop trigger).
161  bool required_;
162 
163  // If this is true, then only the rightmost instance of first synonym found
164  // will be taken from text being checked, and additional instances will
165  // inhibit trigger because repetition actually changes meaning. If false,
166  // then all instances of all synonyms are taken (repetition is meaningless).
167  bool match_once_;
168 
169  // The set of interchangeable alternative representations for this group:
170  // when trying to clear browsing data, a user may think of 'erase', 'clear',
171  // 'delete', etc. Even though these are not strictly synonymous in natural
172  // language, they are considered equivalent within the context of intention
173  // to perform this Pedal's action.
174  std::vector<TokenSequence> synonyms_;
175  };
