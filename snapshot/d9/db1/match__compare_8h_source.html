13 // This class implements a special version of AutocompleteMatch::MoreRelevant
14 // that allows matches of particular types to be demoted in AutocompleteResult.
15 template <class Match>
16 class CompareWithDemoteByType {
17  public:
18  CompareWithDemoteByType(PageClassification page_classification) {
19  OmniboxFieldTrial::GetDemotionsByType(page_classification, &demotions_);
20  }
21 
22  // Returns the relevance score of |match| demoted appropriately by
23  // |demotions_by_type_|.
24  int GetDemotedRelevance(const Match& match) const {
25  auto demotion_it = demotions_.find(match.type);
26  return (demotion_it == demotions_.end())
27  ? match.relevance
28  : (match.relevance * demotion_it->second);
29  }
30 
31  // Comparison function.
32  bool operator()(const Match& elem1, const Match& elem2) {
33  // Compute demoted relevance scores for each match.
34  const int demoted_relevance1 = GetDemotedRelevance(elem1);
35  const int demoted_relevance2 = GetDemotedRelevance(elem2);
36  if (demoted_relevance1 != demoted_relevance2) {
37  // Greater relevance should come first.
38  return demoted_relevance1 > demoted_relevance2;
39  }
40  // For equal-relevance matches, we sort alphabetically, so that providers
41  // who return multiple elements at the same priority get a "stable" sort
42  // across multiple updates.
43  return elem1.contents < elem2.contents;
44  }
45 
46  private:
47  OmniboxFieldTrial::DemotionMultipliers demotions_;
48 };

50 template <class Match>
51 class DestinationSort {
52  public:
53  DestinationSort(PageClassification page_classification)
54  : demote_by_type_(page_classification) {}
55  bool operator()(const Match& elem1, const Match& elem2) {
56  // Sort identical destination_urls together.
57  // Place the most relevant matches first, so that when we call
58  // std::unique(), these are the ones that get preserved.
59  return (elem1.stripped_destination_url == elem2.stripped_destination_url)
60  ? demote_by_type_(elem1, elem2)
61  : (elem1.stripped_destination_url <
62  elem2.stripped_destination_url);
63  }
64 
65  private:
66  CompareWithDemoteByType<Match> demote_by_type_;
67 };
