37 // How history autocomplete works
38 // ==============================
39 //
40 // Read down this diagram for temporal ordering.
41 //
42 // Main thread History thread
43 // ----------- --------------
44 // AutocompleteController::Start
45 // -> HistoryURLProvider::Start
46 // -> VerbatimMatchForInput
47 // [params_ allocated]
48 // -> DoAutocomplete (for inline autocomplete)
49 // -> URLDatabase::AutocompleteForPrefix (on in-memory DB)
50 // -> HistoryService::ScheduleAutocomplete
51 // (return to controller) ----
52 // /
53 // HistoryBackend::ScheduleAutocomplete
54 // -> HistoryURLProvider::ExecuteWithDB
55 // -> DoAutocomplete
56 // -> URLDatabase::AutocompleteForPrefix
57 // /
58 // HistoryService::QueryComplete
59 // [params_ destroyed]
60 // -> AutocompleteProviderListener::OnProviderUpdate
61 //
62 // The autocomplete controller calls us, and must be called back, on the main
63 // thread. When called, we run two autocomplete passes. The first pass runs
64 // synchronously on the main thread and queries the in-memory URL database.
65 // This pass promotes matches for inline autocomplete if applicable. We do
66 // this synchronously so that users get consistent behavior when they type
67 // quickly and hit enter, no matter how loaded the main history database is.
68 // Doing this synchronously also prevents inline autocomplete from being
69 // "flickery" in the AutocompleteEdit. Because the in-memory DB does not have
70 // redirect data, results other than the top match might change between the
71 // two passes, so we can't just decide to use this pass' matches as the final
72 // results.
73 //
74 // The second autocomplete pass uses the full history database, which must be
75 // queried on the history thread. Start() asks the history service schedule to
76 // callback on the history thread with a pointer to the main database. When we
77 // are done doing queries, we schedule a task on the main thread that notifies
78 // the AutocompleteController that we're done.
79 //
80 // The communication between these threads is done using a
81 // HistoryURLProviderParams object. This is allocated in the main thread, and
82 // normally deleted in QueryComplete(). So that both autocomplete passes can
83 // use the same code, we also use this to hold results during the first
84 // autocomplete pass.
85 //
86 // While the second pass is running, the AutocompleteController may cancel the
87 // request. This can happen frequently when the user is typing quickly. In
88 // this case, the main thread sets params_->cancel, which the background thread
89 // checks periodically. If it finds the flag set, it stops what it's doing
90 // immediately and calls back to the main thread. (We don't delete the params
91 // on the history thread, because we should only do that when we can safely
92 // NULL out params_, and that must be done on the main thread.)
93 
94 // Used to communicate autocomplete parameters between threads via the history
95 // service.
96 struct HistoryURLProviderParams {
97  // See comments on |promote_type| below.
98  enum PromoteType {
99  WHAT_YOU_TYPED_MATCH,
100  FRONT_HISTORY_MATCH,
101  NEITHER,

105  const AutocompleteInput& input_before_fixup,

115  // Estimates dynamic memory usage.
116  // See base/trace_event/memory_usage_estimator.h for more info.
117  size_t EstimateMemoryUsage() const;

119  const scoped_refptr<base::SequencedTaskRunner> origin_task_runner;
120 
121  // A copy of the autocomplete input. We need the copy since this object will
122  // live beyond the original query while it runs on the history thread.
123  AutocompleteInput input;
124  // |input_before_fixup| is needed for invoking
125  // |AutocompleteMatch::SetAllowedToBeDefault| which considers
126  // trailing input whitespaces which the fixed up |input| will have trimmed.
127  AutocompleteInput input_before_fixup;
128 
129  // Set when "http://" should be trimmed from the beginning of the URLs.
130  bool trim_http;
131 
132  // A match corresponding to what the user typed.
133  AutocompleteMatch what_you_typed_match;
134 
135  // Set by the main thread to cancel this request. If this flag is set when
136  // the query runs, the query will be abandoned. This allows us to avoid
137  // running queries that are no longer needed. Since we don't care if we run
138  // the extra queries, the lack of signaling is not a problem.
139  base::AtomicFlag cancel_flag;
140 
141  // Set by ExecuteWithDB() on the history thread when the query could not be
142  // performed because the history system failed to properly init the database.
143  // If this is set when the main thread is called back, it avoids changing
144  // |matches_| at all, so it won't delete the default match Start() creates.
145  bool failed;
146 
147  // List of matches written by DoAutocomplete(). Upon its return the provider
148  // converts this list to ACMatches and places them in |matches_|.
149  history::HistoryMatches matches;
150 
151  // True if the suggestion for exactly what the user typed appears as a known
152  // URL in the user's history. In this case, this will also be the first match
153  // in |matches|.
154  //
155  // NOTE: There are some complications related to keeping things consistent
156  // between passes and how we deal with intranet URLs, which are too complex to
157  // explain here; see the implementations of DoAutocomplete() and
158  // FixupExactSuggestion() for specific comments.
159  bool exact_suggestion_is_in_history;
160 
161  // Tells the provider whether to promote the what you typed match, the first
162  // element of |matches|, or neither as the first AutocompleteMatch. If
163  // |exact_suggestion_is_in_history| is true (and thus "the what you typed
164  // match" and "the first element of |matches|" represent the same thing), this
165  // will be set to WHAT_YOU_TYPED_MATCH.
166  //
167  // NOTE: The second pass of DoAutocomplete() checks what the first pass set
168  // this to. See comments in DoAutocomplete().
169  PromoteType promote_type;
170 
171  // True if |what_you_typed_match| is eligible for display. If this is true,
172  // PromoteMatchesIfNecessary() may choose to place |what_you_typed_match| on
173  // |matches_| even when |promote_type| is not WHAT_YOU_TYPED_MATCH.
174  bool have_what_you_typed_match;
175 
176  // The default search provider and search terms data necessary to cull results
177  // that correspond to searches (on the default engine). These can only be
178  // obtained on the UI thread, so we have to copy them into here to pass them
179  // to the history thread. We use a std::unique_ptr<TemplateURL> for the DSP
180  // since TemplateURLs can't be copied by value.
181  std::unique_ptr<TemplateURL> default_search_provider;
182  // Similarly, we use a std::unique_ptr<SearchTermsData> so that we can store a
183  // snapshot of the SearchTermsData accessible from the history thread.
184  std::unique_ptr<SearchTermsData> search_terms_data;
185 
186  // True if the user is allowed to delete browser history. Stored here because
187  // we aren't allowed to read user preferences from the History sequence.
188  const bool allow_deleting_browser_history;
189 };

343  // Params for the current query. The provider should not free this directly;

356  bool search_url_database_;
357
