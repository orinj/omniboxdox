13 // An autocomplete provider that supplies the Verbatim Match for websites.
14 // The match is provided when the user focuses the omnibox, or if the user
15 // clears the contents of the Omnibox while visiting a website.
16 // This provider is intended for use on all platforms that always need the
17 // verbatim match on focus, like the Android Search Ready Omnibox.
18 // Note that the verbatim match may be also supplied by other providers,
19 // eg. ZeroSuggestProvider, but it will be deduplicated and the higher score
20 // from this provider will be used to boost it up to the top.
21 class ZeroSuggestVerbatimMatchProvider : public AutocompleteProvider {
22  public:
23  explicit ZeroSuggestVerbatimMatchProvider(AutocompleteProviderClient* client);
24 
25  void Start(const AutocompleteInput& input, bool minimal_changes) override;
26  void Stop(bool clear_cached_results, bool due_to_user_inactivity) override;
27 
28  private:

30  AutocompleteProviderClient* const client_{nullptr};
31 };
