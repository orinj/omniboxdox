28 class OmniboxPopupModel {
29  public:
30  // Directions for stepping through selections. These may apply for going
31  // up/down by lines or cycling left/right through states within a line.
32  enum Direction { kForward, kBackward };
33 
34  // When changing selections, these are the possible stepping behaviors.
35  enum Step {
36  // Step by an entire line regardless of line state. Usually used for the
37  // Up and Down arrow keys.
38  kWholeLine,
39 
40  // Step by a state if another one is available on the current line;
41  // otherwise step by line. Usually used for the Tab and Shift+Tab keys.
42  kStateOrLine,
43 
44  // Step across all lines to the first or last line. Usually used for the
45  // PgUp and PgDn keys.
46  kAllLines
47  };
48 
49  // The sentinel value for Selection::line which means no line is selected.
50  static const size_t kNoMatch;
51 
52  // See |Selection::state| below for details. The numeric values are to aid
53  // comparison only. They are not persisted anywhere and can be freely changed.
54  enum LineState {
55  // This means the Header above this row is highlighted, and the
56  // header collapse/expand button is focused.
57  FOCUSED_BUTTON_HEADER = 0,
58 
59  // NORMAL means the row is focused, and Enter key navigates to the match.
60  NORMAL = 1,
61 
62  // KEYWORD_MODE state is used when in Keyword mode. If the keyword search
63  // button is enabled, keyword mode is entered when the keyword button is
64  // focused.
65  KEYWORD_MODE = 2,
66 
67  // FOCUSED_BUTTON_TAB_SWITCH state means the Switch Tab button is focused.
68  // Pressing enter will switch to the tab match.
69  FOCUSED_BUTTON_TAB_SWITCH = 3,
70 
71  // FOCUSED_BUTTON_ACTION state means an Action button (such as a Pedal)
72  // is in focus.
73  FOCUSED_BUTTON_ACTION = 4,
74 
75  // FOCUSED_BUTTON_REMOVE_SUGGESTION state means the Remove Suggestion (X)
76  // button is focused. Pressing enter will attempt to remove this suggestion.
77  FOCUSED_BUTTON_REMOVE_SUGGESTION = 5,
78 
79  // Whenever new line state is added, accessibility label for current
80  // selection should be revisited
81  // (OmniboxPopupModel::GetAccessibilityLabelForCurrentSelection).
82  LINE_STATE_MAX_VALUE
83  };
84 
85  struct Selection {
86  // The currently selected line. This is kNoMatch when nothing is selected,
87  // which should only be true when the popup is closed.
88  size_t line;
89 
90  // If the selected line has both a normal match and a keyword match, this
91  // determines whether the normal match (if NORMAL) or the keyword match
92  // (if KEYWORD) is selected. Likewise, if the selected line has a normal
93  // match and a secondary button match, this determines whether the button
94  // match (if FOCUSED_BUTTON_*) is selected.
95  LineState state;
96 
97  explicit Selection(size_t line, LineState state = NORMAL)
98  : line(line), state(state) {}
99 
100  bool operator==(const Selection&) const;
101  bool operator!=(const Selection&) const;
102  bool operator<(const Selection&) const;
103 
104  // Returns true if going to this selection from given |from| selection
105  // results in activation of keyword state when it wasn't active before.
106  bool IsChangeToKeyword(Selection from) const;
107 
108  // Returns true if this selection represents a button being focused.
109  bool IsButtonFocused() const;
110  };

140  // Sets the current selection to |new_selection|. Caller is responsible for
141  // making sure |new_selection| is valid. This assumes the popup is open.
142  //
143  // This will update all state and repaint the necessary parts of the window,
144  // as well as updating the textfield with the new temporary text.
145  //
146  // |reset_to_default| restores the original inline autocompletion.
147  // |force_update_ui| updates the UI even if the selection has not changed.
148  void SetSelection(Selection new_selection,
149  bool reset_to_default = false,
150  bool force_update_ui = false);

152  // We still need to run through the whole SetSelection method, because
153  // changing the line state sometimes requires updating inline autocomplete.
154  void SetSelectedLineState(LineState new_state) {
155  SetSelection(Selection(selected_line(), new_state));
156  }

211  // Triggers the action on |selection| (usually an auxiliary button).

213  // This can't handle all actions currently, and returns false in those cases.

215  // and FOCUSED_BUTTON_ACTION, so is set by default for other use cases.
216  bool TriggerSelectionAction(Selection selection,

218 
219  // This returns the accessibility label for current selection. This is an

223  std::u16string GetAccessibilityLabelForCurrentSelection(
