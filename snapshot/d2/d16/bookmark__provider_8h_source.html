22 // This class is an autocomplete provider which quickly (and synchronously)
23 // provides autocomplete suggestions based on the titles of bookmarks. Page
24 // titles, URLs, and typed and visit counts of the bookmarks are not currently
25 // taken into consideration as those factors will have been used by the
26 // HistoryQuickProvider (HQP) while identifying suggestions.
27 //
28 // TODO(mrossetti): Propose a way to coordinate with the HQP the taking of typed
29 // and visit counts and last visit dates, etc. into consideration while scoring.
30 class BookmarkProvider : public AutocompleteProvider {
31  public:
32  explicit BookmarkProvider(AutocompleteProviderClient* client);
33 
34  BookmarkProvider(const BookmarkProvider&) = delete;
35  BookmarkProvider& operator=(const BookmarkProvider&) = delete;
36 
37  // When |minimal_changes| is true short circuit any additional searching and
38  // leave the previous matches for this provider unchanged, otherwise perform
39  // a complete search for |input| across all bookmark titles.
40  void Start(const AutocompleteInput& input, bool minimal_changes) override;
41 
42  // Sets the BookmarkModel for unit tests.

44  bookmarks::BookmarkModel* bookmark_model) {
45  bookmark_model_ = bookmark_model;
46  }

51  ~BookmarkProvider() override;

53  // Performs the actual matching of |input| over the bookmarks and fills in
54  // |matches_|.
55  void DoAutocomplete(const AutocompleteInput& input);
56 
57  // Get the matches from |bookmark_model_| using the appropriate matching
58  // algorithm, determined by |GetMatchingAlgorithm()|, and path matching
59  // algorithm, determined by the |kBookmarkPaths| base::feature.
60  std::vector<bookmarks::TitledUrlMatch> GetMatchesWithBookmarkPaths(
61  const AutocompleteInput& input,
62  size_t kMaxBookmarkMatches);
63 
64  // There are 2 short bookmark features that determine the matching algorithm
65  // used, i.e. whether input words shorter than 3 chars can prefix match.
66  // 1) |IsShortBookmarkSuggestionsEnabled()| always allows short input word
67  // prefix matching.
68  // 2) |IsShortBookmarkSuggestionsByTotalInputLengthEnabled()| allows short
69  // input word prefix matching only if the input is longer than a threshold
70  // param. This feature also has a counterfactual param.
71  // 3) Otherwise, if both features are disabled (or if the counterfactual param
72  // is true), short input word matching won't be allowed.
73  query_parser::MatchingAlgorithm GetMatchingAlgorithm(AutocompleteInput input);
74 
75  // Calculates the relevance score for |match|.
76  int CalculateBookmarkMatchRelevance(
77  const bookmarks::TitledUrlMatch& match) const;
78 
79  // Removes any URL matches for query parameter keys (if the matching word
80  // starts immediately after a '?' or '&').
81  void RemoveQueryParamKeyMatches(bookmarks::TitledUrlMatch& match);
82 
83  AutocompleteProviderClient* client_;
84  bookmarks::BookmarkModel* bookmark_model_;
85 };
