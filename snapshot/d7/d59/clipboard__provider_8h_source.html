99  // If there is a url copied to the clipboard and accessing it will not show a
100  // clipboard access notification (e.g. iOS 14), use it to create a match.
101  // |read_clipboard_content| will be filled with false if the clipboard didn't
102  // have any content (either because there was none or because accessing it
103  // would have shown a clipboard access notification, and true if there was
104  // content.

106  const AutocompleteInput& input,
107  bool* read_clipboard_content);
108  // If there is text copied to the clipboard and accessing it will not show a

110  // |read_clipboard_content| will be filled with false if the clipboard didn't
111  // have any content (either because there was none or because accessing it
112  // would have shown a clipboard access notification, and true if there was
113  // content.
114  absl::optional<AutocompleteMatch> CreateTextMatch(
115  const AutocompleteInput& input,
116  bool* read_clipboard_content);
117  // If there is an image copied to the clipboard and accessing it will not show
118  // a clipboard access notification (e.g. iOS 14), use it to create a match.
119  // The image match is asynchronous (because constructing the image post data
120  // takes time), so instead of returning an optional match like the other
121  // Create functions, it returns a boolean indicating whether there will be a
122  // match.
123  bool CreateImageMatch(const AutocompleteInput& input);
