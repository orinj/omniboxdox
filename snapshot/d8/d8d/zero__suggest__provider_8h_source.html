34 // Autocomplete provider for searches based on the current URL.
35 //
36 // The controller will call Start() when the user focuses the omnibox. After
37 // construction, the autocomplete controller repeatedly calls Start() with some
38 // user input, each time expecting to receive an updated set of matches.
39 //
40 // TODO(jered): Consider deleting this class and building this functionality
41 // into SearchProvider after dogfood and after we break the association between
42 // omnibox text and suggestions.
43 class ZeroSuggestProvider : public BaseSearchProvider {
44  public:
45  // ZeroSuggestProvider is processing one of the following type of results
46  // at any time. Exposed as public for testing purposes.
47  enum ResultType {
48  NONE,
49 
50  // A remote endpoint (usually the default search provider) is queried for
51  // suggestions. The endpoint is sent the user's authentication state, but
52  // not sent the current URL.
53  REMOTE_NO_URL,
54 
55  // A remote endpoint (usually the default search provider) is queried for
56  // suggestions. The endpoint is sent the user's authentication state and
57  // the current URL.
58  REMOTE_SEND_URL,
59  };
